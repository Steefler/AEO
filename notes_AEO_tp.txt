####Process
ENTITY mux4_1 IS 
PORT (e3, e2, e1, e0 : IN BIT; 

    c : IN INTEGER RANGE 0 TO 3;
    s : OUT BIT);

END mux4_1;

PROCESS("liste de sensibilité nom1, nom2")
BEGIN
         IF (c = 0) THEN s <= e0; 
        ELSIF (c = 1) THEN s <= e1; 
        ELSIF (c = 2) THEN s <= e2; 
	ELSIF (c = 3) THEN s <= e3; 
	END IF;
END PROCESS;

BEGIN
	CASE c IS
		WHEN 0 => s <= e0;
		WHEN 1 => s <= e1;
		WHEN 2 => s <= e2;
		WHEN 3 => s <= e3;
		WHEN OTHERS => ...  pour donner une valeur par défaut à tous les cas restant.
	END CASE;
END PROCESS;
